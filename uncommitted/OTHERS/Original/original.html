// C with OpenMP Grammar
TranslationUnit ::= ( ExternalDeclaration | UnknownCpp | UnknownPragma )+
ExternalDeclaration ::= ( FunctionDefinition | Declaration | ThreadprivateDirective )
FunctionDefinition ::= ( DeclarationSpecifiers )? Declarator ( DeclarationList )? CompoundStatement
Declaration ::= DeclarationSpecifiers ( InitDeclaratorList )? <SEMICOLON>
DeclarationList ::= ( Declaration )+
DeclarationSpecifiers ::= ( StorageClassSpecifier | TypeSpecifier | TypeQualifier )+
StorageClassSpecifier ::= <AUTO>
| <REGISTER>
| <STATIC>
| <EXTERN>
| <TYPEDEF>
TypeSpecifier ::= <VOID>
| <CHAR>
| <SHORT>
| <INT>
| <LONG>
| <FLOAT>
| <DOUBLE>
| <SIGNED>
| <UNSIGNED>
| StructOrUnionSpecifier
| EnumSpecifier
| TypedefName
TypeQualifier ::= <RESTRICT>
| <CONST>
| <VOLATILE>
| <CCONST>
StructOrUnionSpecifier ::= StructOrUnion ( ( <IDENTIFIER> )? <LEFTBRACE> StructDeclarationList <RIGHTBRACE> | <IDENTIFIER> )
StructOrUnion ::= <STRUCT>
| <UNION>
StructDeclarationList ::= ( StructDeclaration )+
InitDeclaratorList ::= InitDeclarator ( <COMMA> InitDeclarator )*
InitDeclarator ::= Declarator ( <OP_ASS> Initializer )?
StructDeclaration ::= SpecifierQualifierList StructDeclaratorList <SEMICOLON>
SpecifierQualifierList ::= ( TypeSpecifier | TypeQualifier )+
StructDeclaratorList ::= StructDeclarator ( <COMMA> StructDeclarator )*
StructDeclarator ::= Declarator ( <COLON> ConstantExpression )?
| <COLON> ConstantExpression
EnumSpecifier ::= <ENUM> ( ( <IDENTIFIER> )? <LEFTBRACE> EnumeratorList <RIGHTBRACE> | <IDENTIFIER> )
EnumeratorList ::= Enumerator ( <COMMA> Enumerator )*
Enumerator ::= <IDENTIFIER> ( <OP_ASS> ConstantExpression )?
Declarator ::= ( Pointer )? DirectDeclarator
DirectDeclarator ::= ( <IDENTIFIER> | <LEFTPAREN> Declarator <RIGHTPAREN> ) ( <LEFTBRACKET> ( ConstantExpression )? <RIGHTBRACKET> | <LEFTPAREN> ParameterTypeList <RIGHTPAREN> | <LEFTPAREN> ( IdentifierList )? <RIGHTPAREN> )*
Pointer ::= <OP_MUL> ( TypeQualifierList )? ( Pointer )?
TypeQualifierList ::= ( TypeQualifier )+
ParameterTypeList ::= ParameterList ( <COMMA> <ELLIPSIS> )?
ParameterList ::= ParameterDeclaration ( <COMMA> ParameterDeclaration )*
ParameterDeclaration ::= DeclarationSpecifiers ( Declarator | ( AbstractDeclarator )? )
IdentifierList ::= <IDENTIFIER> ( <COMMA> <IDENTIFIER> )*
Initializer ::= AssignmentExpression
| <LEFTBRACE> InitializerList ( <COMMA> )? <RIGHTBRACE>
InitializerList ::= Initializer ( <COMMA> Initializer )*
TypeName ::= SpecifierQualifierList ( AbstractDeclarator )?
AbstractDeclarator ::= ( Pointer ( DirectAbstractDeclarator )? | DirectAbstractDeclarator )
DirectAbstractDeclarator ::= ( <LEFTPAREN> AbstractDeclarator <RIGHTPAREN> | <LEFTBRACKET> ( ConstantExpression )? <RIGHTBRACKET> | <LEFTPAREN> ( ParameterTypeList )? <RIGHTPAREN> ) ( <LEFTBRACKET> ( ConstantExpression )? <RIGHTBRACKET> | <LEFTPAREN> ( ParameterTypeList )? <RIGHTPAREN> )*
TypedefName ::= <IDENTIFIER>
Statement ::= ( LabeledStatement | ExpressionStatement | CompoundStatement | SelectionStatement | IterationStatement | JumpStatement | UnknownPragma | OmpConstruct | OmpDirective | UnknownCpp )
UnknownCpp ::= <CROSSBAR> <UNKNOWN_CPP>
OmpEol ::= <OMP_CR>
| <OMP_NL>
OmpConstruct ::= OmpPragma ( ParallelConstruct | ParallelForConstruct | ParallelSectionsConstruct | ForConstruct | SectionsConstruct | SingleConstruct | MasterConstruct | CriticalConstruct | AtomicConstruct | OrderedConstruct )
OmpDirective ::= OmpPragma ( BarrierDirective | FlushDirective )
ParallelConstruct ::= ParallelDirective Statement
OmpPragma ::= <CROSSBAR> <PRAGMA> <OMP>
UnknownPragma ::= <CROSSBAR> <PRAGMA> <UNKNOWN_CPP>
ParallelDirective ::= <PARALLEL> ( UniqueParallelClause | DataClause )* OmpEol
UniqueParallelClause ::= <IF> <LEFTPAREN> Expression <RIGHTPAREN>
ForConstruct ::= ForDirective OmpForHeader Statement
ForDirective ::= <FOR> ( UniqueForClause | DataClause | <NOWAIT> )* OmpEol
UniqueForClause ::= <ORDERED>
| <SCHEDULE> <LEFTPAREN> ScheduleKind ( <COMMA> Expression )? <RIGHTPAREN>
ScheduleKind ::= <STATIC>
| <DYNAMIC>
| <GUIDED>
| <RUNTIME>
OmpForHeader ::= <FOR> <LEFTPAREN> OmpForInitExpression <SEMICOLON> OmpForCondition <SEMICOLON> OmpForReinitExpression <RIGHTPAREN>
OmpForInitExpression ::= <IDENTIFIER> <OP_ASS> Expression
OmpForCondition ::= <IDENTIFIER> ( <OP_LT> | <OP_LE> | <OP_GT> | <OP_GE> ) Expression
OmpForReinitExpression ::= ( <OP_INCR> | <OP_DECR> ) <IDENTIFIER>
| <IDENTIFIER> ( <OP_INCR> | <OP_DECR> )
| <IDENTIFIER> ( <OP_ADDASS> | <OP_SUBASS> ) Expression
| <IDENTIFIER> <OP_ASS> ( <IDENTIFIER> ( <OP_ADD> | <OP_SUB> ) AdditiveExpression | MultiplicativeExpression <OP_ADD> <IDENTIFIER> )
SectionsConstruct ::= <SECTIONS> ( <NOWAIT> | DataClause )* OmpEol SectionsScope
SectionsScope ::= <LEFTBRACE> ( Statement )? ( OmpPragma <SECTION> OmpEol Statement )* <RIGHTBRACE>
SingleConstruct ::= <SINGLE> ( <NOWAIT> | DataClause )* OmpEol Statement
ParallelForConstruct ::= <PARALLEL> <FOR> ( UniqueParallelClause | UniqueForClause | DataClause )* OmpEol OmpForHeader Statement
ParallelSectionsConstruct ::= <PARALLEL> <SECTIONS> ( UniqueParallelClause | DataClause )* OmpEol SectionsScope
MasterConstruct ::= <MASTER> OmpEol Statement
CriticalConstruct ::= <CRITICAL> ( <LEFTPAREN> <IDENTIFIER> <RIGHTPAREN> )? OmpEol Statement
AtomicConstruct ::= <ATOMIC> OmpEol ExpressionStatement
FlushDirective ::= <FLUSH> ( <LEFTPAREN> VariableList <RIGHTPAREN> )? OmpEol
OrderedConstruct ::= <ORDERED> OmpEol Statement
BarrierDirective ::= <BARRIER> OmpEol
ThreadprivateDirective ::= OmpPragma <THREADPRIVATE> <LEFTPAREN> VariableList <RIGHTPAREN> OmpEol
DataClause ::= <PRIVATE> <LEFTPAREN> VariableList <RIGHTPAREN>
| <FIRSTPRIVATE> <LEFTPAREN> VariableList <RIGHTPAREN>
| <LASTPRIVATE> <LEFTPAREN> VariableList <RIGHTPAREN>
| <SHARED> <LEFTPAREN> VariableList <RIGHTPAREN>
| <COPYIN> <LEFTPAREN> VariableList <RIGHTPAREN>
| <DFLT> <LEFTPAREN> ( <SHARED> | <NONE> ) <RIGHTPAREN>
| <REDUCTION> <LEFTPAREN> ReductionOp <COLON> VariableList <RIGHTPAREN>
ReductionOp ::= <OP_ADD>
| <OP_MUL>
| <OP_SUB>
| <OP_BITAND>
| <OP_BITXOR>
| <OP_BITOR>
| <OP_OR>
| <OP_AND>
VariableList ::= <IDENTIFIER> ( <COMMA> <IDENTIFIER> )*
LabeledStatement ::= ( <IDENTIFIER> <COLON> Statement | <CASE> ConstantExpression <COLON> Statement | <DFLT> <COLON> Statement )
ExpressionStatement ::= ( Expression )? <SEMICOLON>
CompoundStatement ::= <LEFTBRACE> ( DeclarationList )? ( StatementList )? <RIGHTBRACE>
StatementList ::= ( Statement )+
SelectionStatement ::= ( <IF> <LEFTPAREN> Expression <RIGHTPAREN> Statement ( <ELSE> Statement )? | <SWITCH> <LEFTPAREN> Expression <RIGHTPAREN> Statement )
IterationStatement ::= ( <WHILE> <LEFTPAREN> Expression <RIGHTPAREN> Statement | <DO> Statement <WHILE> <LEFTPAREN> Expression <RIGHTPAREN> <SEMICOLON> | <FOR> <LEFTPAREN> ( Expression )? <SEMICOLON> ( Expression )? <SEMICOLON> ( Expression )? <RIGHTPAREN> Statement )
JumpStatement ::= ( <GOTO> <IDENTIFIER> <SEMICOLON> | <CONTINUE> <SEMICOLON> | <BREAK> <SEMICOLON> | <RETURN> ( Expression )? <SEMICOLON> )
Expression ::= AssignmentExpression ( <COMMA> AssignmentExpression )*
AssignmentExpression ::= UnaryExpression AssignmentOperator AssignmentExpression
| ConditionalExpression
AssignmentOperator ::= ( <OP_ASS> | <OP_MULASS> | <OP_DIVASS> | <OP_MODASS> | <OP_ADDASS> | <OP_SUBASS> | <OP_SLASS> | <OP_SRASS> | <OP_ANDASS> | <OP_XORASS> | <OP_ORASS> )
ConditionalExpression ::= LogicalORExpression ( <QUESTION> Expression <COLON> ConditionalExpression )?
ConstantExpression ::= ConditionalExpression
LogicalORExpression ::= LogicalANDExpression ( <OP_OR> LogicalORExpression )?
LogicalANDExpression ::= InclusiveORExpression ( <OP_AND> LogicalANDExpression )?
InclusiveORExpression ::= ExclusiveORExpression ( <OP_BITOR> InclusiveORExpression )?
ExclusiveORExpression ::= ANDExpression ( <OP_BITXOR> ExclusiveORExpression )?
ANDExpression ::= EqualityExpression ( <OP_BITAND> ANDExpression )?
EqualityExpression ::= RelationalExpression ( ( <OP_EQ> | <OP_NEQ> ) EqualityExpression )?
RelationalExpression ::= ShiftExpression ( ( <OP_LT> | <OP_GT> | <OP_LE> | <OP_GE> ) RelationalExpression )?
ShiftExpression ::= AdditiveExpression ( ( <OP_SL> | <OP_SR> ) ShiftExpression )?
AdditiveExpression ::= MultiplicativeExpression ( ( <OP_ADD> | <OP_SUB> ) AdditiveExpression )?
MultiplicativeExpression ::= CastExpression ( ( <OP_MUL> | <OP_DIV> | <OP_MOD> ) MultiplicativeExpression )?
CastExpression ::= ( <LEFTPAREN> TypeName <RIGHTPAREN> CastExpression | UnaryExpression )
UnaryExpression ::= ( PostfixExpression | <OP_INCR> UnaryExpression | <OP_DECR> UnaryExpression | UnaryOperator CastExpression | <SIZEOF> ( UnaryExpression | <LEFTPAREN> TypeName <RIGHTPAREN> ) )
UnaryOperator ::= ( <OP_BITAND> | <OP_MUL> | <OP_ADD> | <OP_SUB> | <OP_BITNOT> | <OP_NOT> )
PostfixExpression ::= PrimaryExpression ( <LEFTBRACKET> Expression <RIGHTBRACKET> | <LEFTPAREN> ( ArgumentExpressionList )? <RIGHTPAREN> | <DOT> <IDENTIFIER> | <OP_DEREF> <IDENTIFIER> | <OP_INCR> | <OP_DECR> )*
PrimaryExpression ::= ( <IDENTIFIER> | Constant | <LEFTPAREN> Expression <RIGHTPAREN> )
ArgumentExpressionList ::= AssignmentExpression ( <COMMA> AssignmentExpression )*
Constant ::= <INTEGER_LITERAL>
| <FLOATING_POINT_LITERAL>
| <CHARACTER_LITERAL>
| ( <STRING_LITERAL> )+
